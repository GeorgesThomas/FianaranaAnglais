const XP_PER_DAY = 100;

const DAILY_MISSIONS = [
  { id: "complete_lesson", label: "Complete 1 lesson",      xpReward: 50,  completed: false },
  { id: "finish_quiz",     label: "Finish 1 quiz",          xpReward: 40,  completed: false },

const XP_REWARDS = {
  complete_day:    100,
  quiz_perfect:     50,
  speaking_done:    40,
  chat_message:     10,
  vocab_mastered:   80,
};

const CHAT_XP_DAILY_CAP = 50;
const CHAT_XP_KEY = "eng_imm_chat_xp_v1";

function loadChatXPToday() {
  try {
    const saved = JSON.parse(localStorage.getItem(CHAT_XP_KEY) || "{}");
    return saved.date === todayStr() ? (saved.total ?? 0) : 0;
  } catch { return 0; }
}
function saveChatXPToday(total) {
  try { localStorage.setItem(CHAT_XP_KEY, JSON.stringify({ date: todayStr(), total })); } catch {}
}

function canGainXP(action) {
  if (action !== "chat_message") return true;
  return loadChatXPToday() < CHAT_XP_DAILY_CAP;
}

function addXP(currentXP, action) {
  if (!canGainXP(action)) return currentXP;
  const reward = XP_REWARDS[action] ?? 0;
  if (action === "chat_message") {
    const chatToday = loadChatXPToday();
    const allowed = Math.min(reward, CHAT_XP_DAILY_CAP - chatToday);
    saveChatXPToday(chatToday + allowed);
    return currentXP + allowed;
  }
  return currentXP + reward;
}

const LEVELS = [
  { name: "Beginner",     minXP: 0,    maxXP: 99,   emoji: "🌱", color: "#6ab04c" },
  { name: "Learner",      minXP: 100,  maxXP: 299,  emoji: "📖", color: "#f9ca24" },
  { name: "Intermediate", minXP: 300,  maxXP: 599,  emoji: "⚡", color: "#e17055" },
  { name: "Advanced",     minXP: 600,  maxXP: 999,  emoji: "🔥", color: "#d63031" },
  { name: "Master",       minXP: 1000, maxXP: Infinity, emoji: "🏆", color: "#c9a84c" },
];

function getUserLevel(xp) {
  let level = LEVELS[0];
  for (const l of LEVELS) { if (xp >= l.minXP) level = l; }
  return level;
}

// kept for UI internals (progress bar % to next level)
function getLevelInfo(xp) {
  const level = getUserLevel(xp);
  const idx = LEVELS.indexOf(level);
  const next = LEVELS[idx + 1] ?? null;
  const progress = next
    ? Math.round(((xp - level.minXP) / (next.minXP - level.minXP)) * 100)
    : 100;
  return { ...level, xp, next, progress };
}

function getLevelProgress(xp) {
  const currentLevel = getUserLevel(xp);
  const idx = LEVELS.indexOf(currentLevel);
  const nextLevel = LEVELS[idx + 1] ?? null;
  const progressPercent = nextLevel
    ? Math.min(100, Math.round(((xp - currentLevel.minXP) / (nextLevel.minXP - currentLevel.minXP)) * 100))
    : 100;
  return { currentLevel, nextLevel, progressPercent };
}

// ─── LOCK SYSTEM ─────────────────────────────────────────────────────────────
function isDayUnlocked(day, progress) {
  if (day === 1) return true;
  return progress[`day_${day - 1}_done`] === true;
}

const LEVEL_ORDER = LEVELS.map(l => l.name);

function canAccessFeature(feature, levelName) {
  const required = FEATURE_REQUIREMENTS[feature];
  if (!required) return true;

}

                    <button onClick={(e) => { e.stopPropagation(); togglePron(p.id); }} style={{ background: showPronunciation[p.id] ? C.bg : "none", border: `1px solid ${showPronunciation[p.id] ? C.gold : C.border}`, borderRadius: 4, padding: "6px 12px", color: showPronunciation[p.id] ? C.gold : C.textDim, fontSize: 11, fontFamily: "sans-serif", cursor: "pointer", marginTop: 4 }}>
                      🔊 {showPronunciation[p.id] ? p.pronunciation : "Jereo ny feo"}
                    </button>
                  )}
                </div>
              ))}
            </div>
          </div>
        )}

        {/* ── TAB 1: VOCAB ── */}
        {activeTab === 1 && (
          <div>
            <p style={{ color: C.textSub, fontSize: 14, margin: "0 0 20px", fontFamily: "sans-serif" }}>Teny 15 zarazara araka ny lohahevitra. Tsindrio ny karatra hahitana ohatra.</p>
            <div style={{ display: "flex", gap: 8, marginBottom: 24 }}>
              {["work", "social", "travel"].map(theme => (
                <button key={theme} onClick={() => { setVocabTheme(theme); markDone(`vocab_${theme}`); }} style={{
                  padding: "8px 18px", borderRadius: 4,
                  background: vocabTheme === theme ? C.gold : C.bg4,
                  color: vocabTheme === theme ? "#0d0d0d" : C.goldDim,
                  border: `1px solid ${vocabTheme === theme ? C.gold : C.border}`,
                  cursor: "pointer", fontSize: 13, fontFamily: "sans-serif",
                  fontWeight: vocabTheme === theme ? "700" : "400", transition: "all 0.2s",
                }}>{themeLabels[theme]}</button>
              ))}
            </div>
            <div style={{ display: "flex", flexDirection: "column", gap: 10 }}>
              {data.vocabulary[vocabTheme].map((v, i) => (
                <div key={i} onClick={() => toggleFlip(`v-${vocabTheme}-${i}`)} style={{ background: flipped[`v-${vocabTheme}-${i}`] ? C.bg4 : C.bg3, border: `1px solid ${flipped[`v-${vocabTheme}-${i}`] ? C.gold : C.border}`, borderRadius: 8, padding: "18px 22px", cursor: "pointer", transition: "all 0.2s" }}>
                  <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
                    <div>
                      <div style={{ fontSize: 19, color: C.text, fontStyle: "italic" }}>{v.word}</div>
                      <div style={{ fontSize: 13, color: C.textDim, fontFamily: "sans-serif", marginTop: 2 }}>{v.meaning}</div>
                      {flipped[`v-${vocabTheme}-${i}`] && (
                        <div style={{ marginTop: 12, fontSize: 14, color: "#b0a070", fontFamily: "sans-serif", fontStyle: "italic", borderLeft: `2px solid ${C.gold}`, paddingLeft: 12 }}>"{v.example}"</div>
                      )}
                    </div>
                    <span style={{ fontSize: 18, color: C.goldDark, marginLeft: 16, transform: flipped[`v-${vocabTheme}-${i}`] ? "rotate(45deg)" : "rotate(0deg)", transition: "transform 0.2s" }}>⊕</span>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        {/* ── TAB 2: GRAMMAR ── */}
        {activeTab === 2 && (
          <FeatureLock feature="advanced_grammar">
          <div>
            <p style={{ color: C.textSub, fontSize: 14, margin: "0 0 24px", fontFamily: "sans-serif" }}>Fitsipika 5 hanampiana anao hiteny voajanahary.</p>
            <div style={{ display: "flex", flexDirection: "column", gap: 18 }}>
              {data.grammar.map((g, i) => (
                <div key={i} style={{ background: C.bg3, border: `1px solid ${C.border}`, borderRadius: 8, padding: "22px 24px" }}>
                  <div style={{ fontSize: 11, fontFamily: "sans-serif", letterSpacing: "0.2em", color: C.gold, textTransform: "uppercase", marginBottom: 8 }}>Fitsipika {i + 1}</div>
                  <div style={{ fontSize: 18, color: C.text, fontStyle: "italic", marginBottom: 12 }}>{g.point}</div>
                  <div style={{ fontSize: 13, color: C.textSub, fontFamily: "sans-serif", marginBottom: 16, lineHeight: 1.6, padding: "10px 14px", background: C.bg, borderRadius: 4 }}>💡 {g.rule}</div>
                  <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
                    {g.examples.map((ex, j) => (
                      <div key={j} style={{ fontSize: 13, color: "#b0a070", fontFamily: "sans-serif", fontStyle: "italic", paddingLeft: 14, borderLeft: `2px solid ${C.border}` }}>{ex}</div>
                    ))}
                  </div>
                </div>
              ))}
            </div>
          </div>
          </FeatureLock>
        )}

        {/* ── TAB 3: SPEAKING ── */}
        {activeTab === 3 && (
          <FeatureLock feature="speaking">
          <div>
            <p style={{ color: C.textSub, fontSize: 14, margin: "0 0 8px", fontFamily: "sans-serif" }}>Vakio amin'ny feo mafy io andian-teny 3 minitra io. Reko ny tenanao, avy eo mihainoa!</p>
            <p style={{ color: C.textDim, fontSize: 13, margin: "0 0 24px", fontFamily: "sans-serif" }}>Izany no fiofanana tena izy.</p>
            <div style={{ background: C.bg3, border: `1px solid ${C.border}`, borderRadius: 8, padding: "24px", marginBottom: 20 }}>
              <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 16 }}>
                <div style={{ fontSize: 11, fontFamily: "sans-serif", letterSpacing: "0.2em", color: C.gold, textTransform: "uppercase" }}>Fiofanana — 3 Minitra</div>
                <button onClick={handlePlay} style={{ background: isPlaying ? C.bg2 : C.gold, color: isPlaying ? C.gold : "#0d0d0d", border: `1px solid ${isPlaying ? C.gold : "transparent"}`, borderRadius: 4, padding: "8px 18px", fontSize: 12, fontFamily: "sans-serif", fontWeight: "700", cursor: "pointer" }}>
                  {isPlaying ? "⏹ AJANONA" : "▶ ATOMBOHY"}
                </button>
              </div>
              {(isPlaying || scriptProgress > 0) && (
                <div style={{ marginBottom: 16 }}>
                  <div style={{ height: 4, background: C.bg4, borderRadius: 2, overflow: "hidden" }}>
                    <div style={{ height: "100%", background: C.gold, borderRadius: 2, width: `${scriptProgress}%`, transition: "width 0.5s" }} />
                  </div>
                  <div style={{ fontSize: 11, color: C.textDim, fontFamily: "sans-serif", marginTop: 6 }}>
                    {scriptProgress < 100 ? `${Math.round(scriptProgress * 1.8)}s` : "✓ Vita! Ahoana ny fihevitrao?"}
                  </div>
                </div>
              )}
              <div style={{ fontSize: 15, color: "#d4c898", lineHeight: 1.9, fontStyle: "italic", whiteSpace: "pre-wrap" }}>{data.speakingScript}</div>
            </div>
            <div style={{ background: C.bg, border: `1px solid ${C.border}`, borderRadius: 6, padding: "16px 20px" }}>
              <div style={{ fontSize: 12, color: C.gold, fontFamily: "sans-serif", marginBottom: 8, letterSpacing: "0.1em" }}>TOROLALANA</div>
              {data.speakingTips.map((tip, i) => (
                <div key={i} style={{ fontSize: 13, color: C.goldDim, fontFamily: "sans-serif", padding: "5px 0", borderBottom: i < data.speakingTips.length - 1 ? `1px solid ${C.borderDim}` : "none" }}>
                  {i + 1}. {tip}
                </div>
              ))}
            </div>
          </div>
          </FeatureLock>
        )}

        {/* ── TAB 4: QUIZ ── */}
        {activeTab === 4 && (
          <div>
            <p style={{ color: C.textSub, fontSize: 14, margin: "0 0 24px", fontFamily: "sans-serif" }}>Fanontaniana 8. Aza mitazana ny fanamarinana. Jereo izay niditra an-tsaina.</p>
            <div style={{ display: "flex", flexDirection: "column", gap: 18 }}>
              {data.quiz.map((q, qi) => (
                <div key={qi} style={{ background: C.bg3, border: `1px solid ${C.border}`, borderRadius: 8, padding: "20px 22px" }}>
                  <div style={{ fontSize: 11, color: C.textDim, fontFamily: "sans-serif", letterSpacing: "0.15em", marginBottom: 10 }}>F{qi + 1}</div>
                  <div style={{ fontSize: 15, color: C.text, marginBottom: 14, lineHeight: 1.5 }}>{q.q}</div>
                  <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
                    {q.options.map((opt, ai) => {
                      const selected = quizAnswers[qi] === ai;
                      const correct = quizSubmitted && ai === q.answer;
                      const wrong = quizSubmitted && selected && ai !== q.answer;
                      return (
                        <button key={ai} onClick={() => handleQuizAnswer(qi, ai)} style={{
                          background: correct ? C.greenBg : wrong ? C.redBg : selected ? C.bg4 : C.bg,
                          border: `1px solid ${correct ? C.greenBorder : wrong ? C.redBorder : selected ? C.gold : C.border}`,
                          borderRadius: 5, padding: "11px 16px",
                          color: correct ? C.green : wrong ? C.red : selected ? "#f5e8c0" : C.goldDim,
                          textAlign: "left", cursor: quizSubmitted ? "default" : "pointer",
                          fontSize: 13, fontFamily: "sans-serif", transition: "all 0.15s",
                        }}>
                          {correct && "✓ "}{wrong && "✗ "}{opt}
                        </button>
                      );
                    })}
                  </div>
                </div>
              ))}
            </div>
            {!quizSubmitted ? (
              <button onClick={() => { if (Object.keys(quizAnswers).length === data.quiz.length) { setQuizSubmitted(true); markDone("quiz_main"); } }} style={{ marginTop: 24, width: "100%", padding: "16px", background: Object.keys(quizAnswers).length === data.quiz.length ? C.gold : C.bg4, color: Object.keys(quizAnswers).length === data.quiz.length ? "#0d0d0d" : "#3a2e1a", border: "none", borderRadius: 6, fontSize: 14, fontFamily: "sans-serif", fontWeight: "700", cursor: "pointer", transition: "all 0.2s" }}>
                {Object.keys(quizAnswers).length < data.quiz.length ? `Valío ny fanontaniana ${data.quiz.length} rehetra aloha` : "ALEFASO NY VALINY"}
              </button>
            ) : (
              <div style={{ marginTop: 24, background: isPerfectScore ? "#0f1e08" : C.bg4, border: `1px solid ${isPerfectScore ? C.green : C.gold}`, borderRadius: 8, padding: "24px", textAlign: "center" }}>
                <div style={{ fontSize: 42, marginBottom: 8 }}>{isPerfectScore ? "🌟" : score >= 5 ? "💪" : "📖"}</div>
                <div style={{ fontSize: 28, color: isPerfectScore ? C.green : C.gold, fontStyle: "italic", marginBottom: 6 }}>{score} / {data.quiz.length}</div>
                {isPerfectScore ? (
                  <div>
                    <div style={{ fontSize: 14, color: C.green, fontFamily: "sans-serif", marginBottom: 16 }}>
                      Tonga lafatra! Misokatra ny andro manaraka! +{XP_PER_DAY} XP 🎉
                    </div>
                    <button onClick={() => {
                      // Award for current day based on quiz (uses day 1 as the main quiz represents day content)
                      // Find first incomplete day to award — or use selectedDay if you tie quiz to day
                      const firstIncomplete = DAYS.find(d => isDayUnlocked(d, progress) && !progress[`day_${d}_done`]);
                      if (firstIncomplete) markDayDone(firstIncomplete);
                      setQuizSubmitted(false); setQuizAnswers({});
                    }} style={{ background: C.green, border: "none", borderRadius: 6, padding: "10px 24px", color: "#0d0d0d", fontSize: 13, fontFamily: "sans-serif", fontWeight: "700", cursor: "pointer" }}>
                      🏅 Hanaiky ny latsaka
                    </button>
                  </div>
                ) : (
                  <div>
                    <div style={{ fontSize: 14, color: C.textSub, fontFamily: "sans-serif", marginBottom: 16 }}>
                      Mila 100% ianao mba hahazo ny andro manaraka. Andramo indray!
                    </div>
                    <button onClick={() => { setQuizSubmitted(false); setQuizAnswers({}); }} style={{ marginTop: 0, background: "none", border: `1px solid ${C.border}`, color: C.goldDim, padding: "8px 20px", borderRadius: 4, fontSize: 12, fontFamily: "sans-serif", cursor: "pointer" }}>Andramo indray</button>
                  </div>
                )}
              </div>
            )}
          </div>
        )}

        {/* ── TAB 5: AI CHAT ── */}
        {activeTab === 5 && (
          <FeatureLock feature="chat_ai">
          <div>
            {!chatTopic ? (
              <div>
                <p style={{ color: C.textSub, fontSize: 14, margin: "0 0 8px", fontFamily: "sans-serif" }}>Miresaka anglisy mivantana amin'i Jordan avy any New York. Raha diso ny fitsipika, hisy fanitsiana malefaka.</p>
                <p style={{ color: C.textDim, fontSize: 13, margin: "0 0 24px", fontFamily: "sans-serif" }}>Safidio ny lohahevitra — 6 amin'izao!</p>
                <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 10 }}>
                  {TOPICS.map(topic => (
                    <button key={topic.id} onClick={() => startChat(topic)} style={{ background: C.bg3, border: `1px solid ${C.border}`, borderRadius: 8, padding: "18px 16px", cursor: "pointer", textAlign: "left", transition: "all 0.2s" }}
                      onMouseEnter={e => e.currentTarget.style.borderColor = C.gold}
                      onMouseLeave={e => e.currentTarget.style.borderColor = C.border}
                    >
                      <div style={{ fontSize: 22, marginBottom: 6 }}>{topic.emoji}</div>
                      <div style={{ fontSize: 15, color: C.text, fontStyle: "italic", marginBottom: 4 }}>{topic.label}</div>
                      <div style={{ fontSize: 11, color: C.textDim, fontFamily: "sans-serif" }}>{topic.desc}</div>
                    </button>
                  ))}
                </div>
              </div>
            ) : (
              <div>
                <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 16 }}>
                  <div>
                    <div style={{ fontSize: 11, color: C.gold, fontFamily: "sans-serif", letterSpacing: "0.2em", textTransform: "uppercase" }}>Resaka amin'i Jordan — {chatTopic.emoji} {chatTopic.label}</div>
                    <div style={{ fontSize: 12, color: C.textDim, fontFamily: "sans-serif", marginTop: 3 }}>Mpiteny anglisy avy any New York · Manakarama kafe</div>
                  </div>
                  <button onClick={() => { setChatTopic(null); setMessages([]); }} style={{ background: "none", border: `1px solid ${C.border}`, borderRadius: 4, color: C.textDim, padding: "6px 14px", fontSize: 11, fontFamily: "sans-serif", cursor: "pointer" }}>← Hiverina</button>
                </div>
                <div style={{ background: "#090905", border: `1px solid ${C.borderDim}`, borderRadius: 6, padding: "10px 14px", marginBottom: 14, fontSize: 12, color: C.textDim, fontFamily: "sans-serif" }}>
                  💡 Raha diso ny fitsipika, holazain'i Jordan hoe <span style={{ color: C.gold }}>"That sounds good! Just one small thing…"</span>
                </div>
                <div style={{ background: "#0a0a05", border: `1px solid ${C.borderDim}`, borderRadius: 8, padding: "16px", minHeight: 300, maxHeight: 400, overflowY: "auto", marginBottom: 12, display: "flex", flexDirection: "column", gap: 12 }}>
                  {visibleMessages.length === 0 && chatLoading && (
                    <div style={{ color: C.textDim, fontFamily: "sans-serif", fontSize: 13, fontStyle: "italic" }}>Jordan dia miomana… ☕</div>
                  )}
                  {visibleMessages.map((msg, i) => (
                    <div key={i} style={{ display: "flex", flexDirection: msg.role === "user" ? "row-reverse" : "row", gap: 10, alignItems: "flex-end" }}>
                      {msg.role === "assistant" && (
                        <div style={{ width: 32, height: 32, borderRadius: "50%", background: C.bg4, border: `1px solid ${C.goldDark}`, display: "flex", alignItems: "center", justifyContent: "center", fontSize: 14, flexShrink: 0 }}>☕</div>
                      )}
                      <div style={{ maxWidth: "78%", padding: "11px 15px", borderRadius: 10, background: msg.role === "user" ? C.bg4 : C.bg3, border: `1px solid ${msg.role === "user" ? C.gold : C.border}`, fontSize: 14, color: msg.role === "user" ? "#f5e8c0" : "#d4c898", fontFamily: "sans-serif", lineHeight: 1.6 }}>
                        {msg.content}
                      </div>
                    </div>
                  ))}
                  {chatLoading && visibleMessages.length > 0 && (
                    <div style={{ display: "flex", gap: 10, alignItems: "flex-end" }}>
                      <div style={{ width: 32, height: 32, borderRadius: "50%", background: C.bg4, border: `1px solid ${C.goldDark}`, display: "flex", alignItems: "center", justifyContent: "center", fontSize: 14 }}>☕</div>
                      <div style={{ padding: "12px 16px", borderRadius: 10, background: C.bg3, border: `1px solid ${C.border}`, fontSize: 14, color: C.textSub, fontFamily: "sans-serif", fontStyle: "italic" }}>Jordan dia manoratra…</div>
                    </div>
                  )}
                  <div ref={chatEndRef} />
                </div>
                <div style={{ display: "flex", gap: 8 }}>
                  <input value={userInput} onChange={e => setUserInput(e.target.value)} onKeyDown={e => e.key === "Enter" && !e.shiftKey && sendMessage()}
                    placeholder="Soraty ny valiny eto…"
                    style={{ flex: 1, background: C.bg3, border: `1px solid ${C.border}`, borderRadius: 6, padding: "12px 16px", color: C.text, fontSize: 14, fontFamily: "sans-serif", outline: "none" }}
                  />
                  <button onClick={sendMessage} disabled={chatLoading || !userInput.trim()} style={{ background: userInput.trim() && !chatLoading ? C.gold : C.bg4, color: userInput.trim() && !chatLoading ? "#0d0d0d" : C.textDim, border: "none", borderRadius: 6, padding: "12px 20px", fontSize: 13, fontFamily: "sans-serif", fontWeight: "700", cursor: "pointer", transition: "all 0.2s" }}>
                    {chatLoading ? "…" : "↑"}
                  </button>
                </div>
              </div>
            )}
          </div>
          </FeatureLock>
        )}

        {/* ── TAB 6: VOCAB MEMORY (TENY VAOVAO) ── */}
        {activeTab === 6 && <FeatureLock feature="vocab_memory">{renderVocabMemory()}</FeatureLock>}

      </div>
    </div>
  );
}
